class Solution:
    def coinChange(self, coins, amount):
        def dp(rem):
            if rem == 0:
                return 0
            if rem < 0:
                return float('inf')
            best = float('inf')
            for c in coins:
                best = min(best, dp(rem-c)+1)
            return best

        res = dp(amount)
        return res if res != float('inf') else -1

class Solution:
    def coinChange(self, coins, amount):
        memo = {}
        def dp(rem):
            if rem == 0:
                return 0
            if rem < 0:
                return float('inf')
            if rem in memo:
                return memo[rem]
            best = float('inf')
            for c in coins:
                best = min(best, dp(rem-c)+1)
            memo[rem] = best
            return best

        res = dp(amount)
        return res if res != float('inf') else -1
