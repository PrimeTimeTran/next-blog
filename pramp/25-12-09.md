```py
'''
1. Sentence Reverse

Time:   O(n) where n is the length of s
Space:  O(n) due to split list and output string
'''
class Solution:
    def reverseWords(self, s: str) -> str:
        return ' '.join(s.split()[::-1])
```

```python
'''
1. Decode Variations

Time:   O(n) where n is the length of s
Space:  O(n) due to split list and output string
'''
def decodeVariations(s: str) -> int:
    memo = {}
    def dfs(i):
        if i in memo:
            return memo[i]
        if i == len(s):
            return 1
        if s[i] == "0":
            return 0
        count = dfs(i+1)
        if i < len(s) and 10 <= int(s[i:i+2]) <= 26:
            count += dfs(i+2)
        memo[i] = count
        return memo[i]
    return dfs(0)
```

```py
'''
33. Search in Rotated Sorted Array
'''
class Solution:
  def search(self, nums: List[int], target: int) -> int:
    l, r = 0, len(nums) - 1

    while l <= r:
        m = (l+r) // 2

        if nums[m] == target:
            return m
        elif nums[l] <= nums[m]:
            if nums[l] <= target < nums[m]:
                r = m - 1
            else:
                l = m + 1
        else:
            if nums[m] < target <= nums[r]:
                l = m + 1
            else:
                r = m - 1
    return -1
```

```py
'''
1. Toeplitz Matrix

A Toeplitz matrix is a matrix where every left-to-right-descending diagonal has the same element. Given a non-empty matrix arr, write a function that returns true if and only if it is a Toeplitz matrix. The matrix can be any dimensions, not necessarily square.

[[1,2,3,4],
 [5,1,2,3],
 [6,5,1,2]]
'''
def isToeplitz(arr: List[List[int]]) -> bool:
    R, C = len(arr), len(arr[0])
    for r in range(R):
        for c in range(C):
            if r >= 1 and c >= 1:
                if not arr[r-1][c-1] == arr[r][c]:
                    return False
    return True
```
